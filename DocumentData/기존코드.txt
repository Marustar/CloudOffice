def document_upload(request):
    if request.method == 'POST':
        form = DocumentForm(request.POST, request.FILES)
        if form.is_valid():
            form.save()
            document = Document()
            user = Employee.objects.get(Emp_User = request.user)
            document.Doc_ID = 1
            document.Doc_Dept = user.Emp_Dept
            document.Doc_Sender = user.Emp_Name
            document.Doc_Receiver = user.Emp_Name
            document.Doc_Title = "132"
            document.Doc_Type = 3
            document.Doc_State = 2
            document.save()
            return redirect('/document_upload?success_page=true')
        else:
            print(form.errors) 
    else:
        form = DocumentForm()
    return render(request, 'fileupload.html', {'form': form})
